            CREATE TABLE boards (
                idx  INTEGER  PRIMARY KEY  AUTOINCREMENT,
                name   TEXT,
                moderator  TEXT
            );
            CREATE TABLE posts (
                serial_number  INTEGER  PRIMARY KEY  AUTOINCREMENT,
                board   TEXT,
                title   TEXT,
                content TEXT,
                author  TEXT,
                date    DATETIME  DEFAULT CURRENT_TIMESTAMP
            );
            CREATE TABLE comments (
                idx    INTEGER  PRIMARY KEY  AUTOINCREMENT,
                post_sn  INT,
                content  TEXT,
                author   TEXT
            );
